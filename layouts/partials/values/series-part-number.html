{{- /* Quick and dirty way to figure out the article series part number */ -}} {{- $partNumber := 0 -}} {{- if .Parent -}} {{- if eq .Parent.Type "series" -}} {{- range $index, $page := .Parent.Pages -}} {{- if eq $page $.Page -}} {{- $partNumber = (add $index 1 ) -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{ $partNumber }}
